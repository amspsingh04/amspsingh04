## Query pattern (events → team summaries)

```python
import pandas as pd

def team_match_summary(events: pd.DataFrame, match_id: int) -> pd.DataFrame:
    m = events[events["match_id"] == match_id].copy()
    # Pseudocode: explode team_id, filter shots, sum stats
    rows = []
    for team_id, g in m.groupby("team.id"):
        shots = g[g["type"] == "Shot"]
        np_xg = shots["shot"].apply(lambda s: s.get("statsbomb_xg", 0)).sum()
        rows.append({"team_id": team_id, "shots": len(shots), "np_xg": round(np_xg, 2)})
    return pd.DataFrame(rows)
```
